170 	//if $element exist, then, bypass the build
171 	function process(sui,name,data,config,$element){
172 		var componentDef = sui.getComponentDef(name);
173 		
174 		config = buildConfig(componentDef,config);
175 		
176 		var component = instantiateComponent(componentDef);
177 		
178 		//if the config.unique is set, and there is a component with the same name, abort
179 		//TODO: an optimization point would be to add a "sComponentUnique" in the class for data-scomponent that have a confi.unique = true
180 		//      This way, the query below could be ".sComponentUnique [....]" and should speedup the search significantly on UserAgents that supports the getElementsByClassName
181 		if (config.unique && $("[data-scomponent='" + name + "']").length > 0){
182 			return null;
183 		}
184 		
185 		// if there is no element, we invoke the build
186 		if (!$element) {
187 			//Ask the component to build the new $element
188 			$element = invokeBuild(component, data, config);
189 		}
190 		
191 		//if there is an element, then, manage the rendering logic. 
192 		if ($element) {
193 			//make sure we get the jQuery object
194 			$element = $($element);
195 			
196 			bind$element($element, component, data, config);
197 			
198 			//render the element
199 			renderComponent(snow.ui, component, data, config);
200 			
201 			
202 			
203 		} 
204 		
205 		invokePostDisplay(component, data, config);
206 		
207 		return component;		
208 	}

210 	function renderComponent (sui,component,data,config){
211 	
212 		if (config.transition){
213 			var transition = sui.getTransition(config.transition);
214 			
215 			if (transition) {
216 				transition(component,data,config);
217 			}else{
218 				snow.log.error("Transition [" + config.transition + "] not found. Transitions need to be registered via snow.ui.registerTranstion(..) before call.");
219 			}
220 		}
221 		//if no transition remove/show
222 		else{
223 			if (config.replace){
224 				$(config.replace).remove();
225 			}
226 			
227 			//note: if there is no parent, then, the sUI.diplay caller is reponsible to add it
228 			if (config.parent) {
229 				if (config.emptyParent){
230 					$(config.parent).empty();
231 				}
232 				$(config.parent).append(component.$element);
233 			}
234 		}

247 	function instantiateComponent(componentDef){
248 		var component; 
249 		var componentFactory = componentDef.componentFactory;
250 		if (componentFactory ){
251 			// if it is a function, call it, it should return a new component object
252 			if ($.isFunction(componentFactory)) {
253 				component = componentFactory();
254 			}
255 			// if it is a plainObject, then, we clone it (NOTE: We do a one level clone)
256 			else if ($.isPlainObject(componentFactory)){
257 				component = $.extend({},componentFactory);
258 			}else{
259 				snow.log.error("Invalid ComponentFactory for component [" + componentDef.componentName + "]. Only types Function or Object are supported as componentFactory. Empty component will be created.");				
260 			}
261 		}else{
262 			snow.log.error("No ComponentFactory for component [" + componentDef.componentName + "]");
263 		}
264 		
265 		if (component){
266 			component.name = componentDef.name;
267 		}
268 		return component;		
269 	}

319    *
320    * Return the component that this html element belong to. Thi traverse the tree backwards (this html element up to document) to find the closest html element
321    * containing the snow.ui component for this name. 
322    * 
323    * If a componentName is given then it will try to find the given component. 
324    * 
325    * If no componentName is given, then it will return the first component found.
326    * 
327    * For example: 
328    * @example
329    * var myComponent = $(thisDiv).sComponent("myComponent");
330    * 
331    * @param {String} componentName The component name to be match when traversing the tree, if undefined, then, the closestComponent will be return.
332    * 
333    */
334   $.fn.sComponent = function(componentName) {
335 	  
336       // iterate and process each matched element
337 	  	var $componentElement; 
338 		if (componentName) {
339 			$componentElement = $(this).closest("[data-scomponent='" + componentName + "']");
340 		}else{
341 			$componentElement = $(this).closest("[data-scomponent]");
342 		}
343 		
344 		return $componentElement.data("component");
345  
346    };

451 (function() {
452   // Private array of chars to use
453   var CHARS = '0123456789ABCDEFGHIJKLMNOPQRSTUVWXYZabcdefghijklmnopqrstuvwxyz'.split(''); 
454 
455 /**
456  * Create a random id.
457  * <br /><br />
458  * Code from: Math.uuid 2010 Robert Kieffer http://www.broofa.com 
459  * 
460  * 
461  * @example
462  *    snow.util.uuid(); // returns "92329D39-6F5C-4520-ABFC-AAB64544E172"
463  *    snow.util.uuid(15);    // 15 character ID (default base=62), returns "VcydxgltxrVZSTV"
464  *    snow.util.uuid(8, 2);  // returns "01001010"
465  *    
466  * @param {Number} len (optional) length in char of the returned random ID. If absent, the standard UUID format will be returned 
467  * @param {Number} radix (optional) radix of the random number. (Default: 62)
468  */
469   snow.util.uuid =  function (len, radix) {
470     var chars = CHARS, uuid = [];
471     radix = radix || chars.length;
472 
473     if (len) {
474       // Compact form
475       for (var i = 0; i < len; i++) uuid[i] = chars[0 | Math.random()*radix];
476     } else {
477       // rfc4122, version 4 form
478       var r;
479 
480       // rfc4122 requires these characters
481       uuid[8] = uuid[13] = uuid[18] = uuid[23] = '-';
482       uuid[14] = '4';
483 
484       // Fill in random data.  At i==19 set the high bits of clock sequence as
485       // per rfc4122, sec. 4.1.5
486       for (var i = 0; i < 36; i++) {
487         if (!uuid[i]) {
488           r = 0 | Math.random()*16;
489           uuid[i] = chars[(i == 19) ? (r & 0x3) | 0x8 : r];
490         }
491       }
492     }
493 
494     return uuid.join('');
495   };
496 })();

511 	 * @param {Object} a the Array 
512 	 * @param {Object} from the first index to remove from

515 	remove: function(a,from, to) {
516 	  var rest = a.slice((to || from) + 1 || a.length);
517 	  a.length = from < 0 ? a.length + from : from;
518 	  return a.push.apply(a, rest);

521 	/**
522 	 * For a array of object, this will get the first index of the matching prop name/value 
523 	 * return -1 if no match
524 	 * @param {Object} a the Array
525 	 * @param {Object} propName the property name
526 	 * @param {Object} propValue the property value to be matched
527 	 */
528 	getIndex: function(a,propName,propValue){
529 		if (a && propName && typeof propValue != "undefined"){
530 			var i,obj,l = a.length;
531 			for (var i = 0; i < l; i++){
532 				obj = a[i];
533 				if (obj && obj[propName] === propValue){
534 					return i;
535 				}
536 			}
537 		}
538 		return -1;
539 	},

541 	/**
542 	 * Sort an array of object by a propName
543 	 * @param {Object} a the Array
544 	 * @param {Object} propName the property name to be sorted by
545 	 */
546 	sortBy: function(a,propName){
547 		return a.sort(sortByFunc);
548 		function sortByFunc(a, b) {
549 			if (typeof a === "undefined") return -1;
550 			if (typeof b === "undefined") return 1;
551 			
552 		    var x = a[propName];
553 		    var y = b[propName];
554 		    return ((x < y) ? -1 : ((x > y) ? 1 : 0));
555 		}
556 	},

561 	 * @example
562 	 * var myVehicules = [{id:"truck",speed:80},{id:"racecar",speed:200}];

848 	// private Method to compute the ratio
849 	// Note that this will also compute offset to position the element center if the x and y ration 
850 	// are not identical
851 	function computeRatio(){
852 		var w = this.canvas().width; h = this.canvas().height;
853 		
854 		//TODO: probably need to test undefined and null, because 0 should be valid
855 		if (w && h && this._refWidth && this._refHeight){
856 			this._xRatio = w / this._refWidth;
857 			this._yRatio = h / this._refHeight;
858 			//TODO: this probably need to be fixed for not square artifacts
859 			if (this._xRatio <= this._yRatio ){
860 				this._ratio = this._xRatio;
861 				this._xOffset = 0;
862 				this._yOffset = (h - this._refHeight * this._ratio ) / 2;
863 			}else{
864 				this._ratio = this._yRatio;
865 				this._xOffset = (w - this._refWidth * this._ratio ) / 2;
866 				this._yOffset = 0;
867 			}
868 			 
869 			this.scallable = true;
870 		}
871 		
872 	}

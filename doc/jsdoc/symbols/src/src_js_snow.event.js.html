  4     /**
  5      * This event either bind a touchend if available, otherwise, bind a click.
  6      * In future iterations, we might want make the "touch" case a little bit more advanced (checking touchstart/touchend are from the same element and no drag was initiated)
  7      * @param {Function or String} arg0 can be the function or a "delegate" selector string if we want this event to be delegated
  8      * @param {Function} arg1 if arg0 is the del
  9      */
 10     $.fn.sTouch = function(arg0, arg1){
 11         var callback = arg1 || arg0;
 12         var delegate = (arg1) ? arg0 : null;
 13         
 14         var eventName = (snow.ua.hasTouch()) ? "touchend" : "click";
 15 		
 16         return this.each(function(){
 17             var $this = $(this);
 18             
 19             if (delegate) {
 20                 $this.delegate(delegate, eventName, callback);
 21             }
 22             else {
 23                 $this.bind(eventName, callback);
 24             }
 25             
 26             
 27         });
 28         
 29     };
 30     
 31     // samplePlugin default options
 32     $.fn.sTouch.defaults = {};

 37 (function($){
 38 	const SDRAGSTART="sDragStart",SDRAGDRAG="sDragDrag",SDRAGEND="sDragEnd";
 39     /**
 40      *
 41      * Options optional method implementation:
 42      *
 43      *
 44      */
 45     var mouseDragEvents = {
 46         start: "mousedown",
 47         drag: "mousemove",
 48         end: "mouseup"
 49     }
 50     
 51     var touchDragEvents = {
 52         start: "touchstart",
 53         drag: "touchmove",
 54         end: "touchend"
 55     }
 56     
 57     var hasTouch = snow.ua.hasTouch();
 58     
 59     
 60     /**
 61      * Drag event for mouse and touch based devices. 
 62      *
 63      * @param {Object} delegate (optional) if present, this will do a delegate (not implemented yet)
 64      * @param {Object} opts options and handlers
 65      * @param {Function} opts.start=function(event,extra) [optional] will be called when the drag is initiated
 66      * @param {Function} opts.drag=function(event,extra) [optional] will be called for every drag event (either mousemouve or touchmove)
 67      * @param {Function} opts.end=function(event,extra) [optional] called on mouseUp or touch end
 68      * @param {Number} extra.pageX the current pageX 
 69      * @param {Number} extra.pageY the current pageY 
 70      * @param {Number} extra.startPageX the pageX from the drag start
 71      * @param {Number} extra.startPageY the pageY from the drag start
 72      * @param {Number} extra.deltaPageX the delta between the current pageX and the last one (from the previous drag event)
 73      * @param {Number} extra.deltaPageY the delta between the current pageY and the last one (from the previous drag event)
 74      * 
 75      *
 76      * TODO: needs to send jQuery event rather than hardcoding calls to events. <br />
 77      * TODO: needs to find a solution for touch vs mouse move box-bounding difference
 78      */
 79     $.fn.sDrag = function(delegate, opts){
 80         var options = opts || delegate;
 81         var delegate = (opts) ? delegate : null;
 82         
 83         
 84         var dragEvents = (hasTouch) ? touchDragEvents : mouseDragEvents;
 85         
 86         //for now, support the not delegatable way
 87         // iterate and process each matched element
 88         return this.each(function(){
 89             var $this = $(this); // jQuery object for this element
 90             if (delegate == null) {
 91 				(options.start)?$this.bind(SDRAGSTART,options.start):null;
 92 				(options.drag)?$this.bind(SDRAGDRAG,options.drag):null;
 93 				(options.end)?$this.bind(SDRAGEND,options.end):null;
 94 				
 95                 $this.bind(dragEvents.start, function(e){
 96                    handleDragEvent.call(this,e,options);
 97                 });
 98             }else{
 99 				
100 				(options.start)?$this.delegate(delegate,SDRAGSTART,options.start):null;
101 				(options.drag)?$this.delegate(delegate,SDRAGDRAG,options.drag):null;
102 				(options.end)?$this.delegate(delegate,SDRAGEND,options.end):null;				
103 				
104 				$this.delegate(delegate,dragEvents.start,function(e){
105 					handleDragEvent.call(this,e,options);
106 				})
107 			}
108         });
109         
110 		// Handler the event
111 		// "this" of this function will be the element
112         function handleDragEvent(e, options){
113 			var $this = $(this);
114             fixTouchEvent(e);
115             var $handle = $(this);
116             
117             var $document = $(document);
118             var id = "_" + snow.util.uuid(7);
119             
120 			var extra = buildExtra(e,$handle,SDRAGSTART);
121             $this.trigger(SDRAGSTART,[extra]);
122             
123             e.preventDefault();
124             
125             $document.bind(dragEvents.drag + "." + id, function(e){
126                 fixTouchEvent(e);
127 				
128 				extra = buildExtra(e,$handle,SDRAGDRAG);
129             	$this.trigger(SDRAGDRAG,[extra]);
130 				
131                 e.preventDefault();
132             });
133             
134             $document.bind(dragEvents.end + "." + id, function(e){
135                 fixTouchEvent(e);
136                 var extra = buildExtra(e,$handle,SDRAGEND);
137             	$this.trigger(SDRAGEND,[extra]);
138 				
139                 $(document).unbind(dragEvents.drag + "." + id);
140                 $(document).unbind(dragEvents.end + "." + id);
141                 e.preventDefault();
142             });
143         }
144         
145     }
146     
147 	
148 	// assumed that the event has been fixed
149 	function buildExtra(event,$handle,dragType){
150 		var extra = {
151 			eventSource: event,
152 			pageX: event.pageX,
153 			pageY: event.pageY			
154 		};
155 		
156 		var currentState = $handle.data("_sDrag_currentState");
157 		if (!currentState){
158 			currentState = {};
159 			$handle.data("_sDrag_currentState",currentState);
160 		}
161 		
162 		if (dragType === SDRAGSTART){
163 			currentState.startPageX = extra.startPageX = extra.pageX;
164 			currentState.startPageY = extra.startPageY = extra.pageY;
165 			
166 			currentState.lastPageX = currentState.startPageX = extra.startPageX;
167 			currentState.lastPageY = currentState.startPageY = extra.startPageY;
168 		}else if (dragType === SDRAGEND){
169 			// because, on iOs, the touchEnd event does not have the .touches[0].pageX
170 			extra.pageX = currentState.lastPageX;
171 			extra.pageY = currentState.lastPageY;
172 			$handle.data("_sDrag_currentState",null);
173 		}
174 		
175 		extra.startPageX = currentState.startPageX;
176 		extra.startPageY = currentState.startPageY;
177 		extra.deltaPageX = extra.pageX - currentState.lastPageX;
178 		extra.deltaPageY = extra.pageY - currentState.lastPageY;
179 		
180 		currentState.lastPageX = extra.pageX;
181 		currentState.lastPageY = extra.pageY;
182 		return extra;
183 	}
184     
185     /**
186      * First attempt to fix the touchEvent to make it more mouse like
187      * @param {Object} e
188      */
189     function fixTouchEvent(e){
190         if (hasTouch) {
191             var oe = e.originalEvent;
192             if (oe.touches.length > 0) {
193                 e.pageX = oe.touches[0].pageX;
194                 e.pageY = oe.touches[0].pageY;
195             }
196         }
197         
198         return e;
199     }

  9 (function($){
 10 	var hasTouch = snow.ua.hasTouch();
 11 	/**
 12      * if it is a touch device, populate the event.pageX and event.page& from the event.touches[0].pageX/Y
 13      * @param {jQuery Event} e the jquery event object 
 14      */
 15     snow.event.fixTouchEvent = function(e){
 16         if (hasTouch) {
 17             var oe = e.originalEvent;
 18 			
 19             if (oe.touches.length > 0) {
 20                 e.pageX = oe.touches[0].pageX;
 21                 e.pageY = oe.touches[0].pageY;
 22             }
 23         }
 24         
 25         return e;
 26     }
 27     
 28     /**
 29      * Return the event {pageX,pageY} object for a jquery event object (will take the touches[0] if it is a touch event)
 30      * @param {jQuery Event} e the jquery event object
 31      */
 32     snow.event.eventPagePosition = function(e){
 33 		if (e.originalEvent && e.originalEvent.touches){
 34 			pageX = e.originalEvent.touches[0].pageX;
 35 			pageY = e.originalEvent.touches[0].pageY;
 36 		}else{
 37 			pageX = e.pageX;
 38 			pageY = e.pageY;
 39 		}
 40 		return {
 41 			pageX: pageX,
 42 			pageY: pageY
 43 		}
 44     }

 69             }
 70             else {
 71                 $this.bind(eventName, callback);
 72             }
 73

 75         });
 76

139 (function($){
140 	const SDRAGSTART="sdragstart",SDRAGDRAG="sdrag",SDRAGEND="sdragend";
141 	
142 	const SDRAGENTER="sdragenter",SDRAGOVER="sdragover",SDRAGLEAVE="sdragleave",SDROP="sdrop";

146      *
147      * Options optional method implementation:
148      *
149      *

151     var mouseDragEvents = {
152         start: "mousedown",
153         drag: "mousemove",
154         end: "mouseup"
155     }

157     var touchDragEvents = {
158         start: "touchstart",
159         drag: "touchmove",
160         end: "touchend"
161     }

167     /**
168      * Drag event for mouse and touch based user agents. 
169      * 
170      * This fires sdragstart, sdrag, and sdragend events on the dragged element, 
171      * and sdragenter, sdragleave, sdrop on the "overElement" if the opts.draggable is set to true. If opts.draggable is omitted or false, then
172      * only the events on the dragged element will be fired. 
173      * 
174      * See code sample at /test/test_snow.event.02.sDrag.html for more infor 
175      *
176      * @param {String} delegate [optional] if present, this will do a delegate (not implemented yet)
177      * @param {Object}   opts [optional] options and handlers
178      * @param {Boolean}  opts.draggable [default false] tell if the element is draggable
179      * @param {String|Function}   opts.helper [default 'original'] if opts.draggable is true, this will determine the strategy for the drag helper. 
180      *                               Can be 'clone' 'original' or a function
181      * @param {Function} opts.start=function(event,dragExtra) [optional] will be called when the drag is initiated (map to the 'sdragstart' event)
182      * @param {Function} opts.drag=function(event,dragExtra) [optional] will be called for every drag event (map to the 'sdrag' event)
183      * @param {Function} opts.end=function(event,dragExtra) [optional] called on mouseUp or touch end (map to the 'sdragend' event)
184      * @param {Number} dragExtra.pageX the current pageX 
185      * @param {Number} dragExtra.pageY the current pageY 
186      * @param {Number} dragExtra.startPageX the pageX from the drag start
187      * @param {Number} dragExtra.startPageY the pageY from the drag start
188      * @param {Number} dragExtra.deltaPageX the delta between the current pageX and the last one (from the previous drag event)
189      * @param {Number} dragExtra.deltaPageY the delta between the current pageY and the last one (from the previous drag event)
190      * @param {DOMElement} dragExtra.overElement this is for sdrag and sdragend event, and represent the over element (the drop over element) if the opts.draggable was set to true
191      * @param {DOMElement} dragExtra.helperElement this is the element that is used for the drag effect. Could be the same as the draggable element, or a clone or custom element.
192      */
193     $.fn.sDrag = function(delegate, opts){
194         var options = opts || delegate;
195         var delegate = (opts) ? delegate : null;
196         var hasTouch = snow.ua.hasTouch();
197         
198         options = $.extend({},$.fn.sDrag.defaults,options);
199         
200         var dragEvents = (hasTouch) ? touchDragEvents : mouseDragEvents;
201         
202         //for now, support the not delegatable way
203         // iterate and process each matched element
204         return this.each(function(){
205             var $this = $(this); // jQuery object for this element
206             if (delegate == null) {
207 				(options.start)?$this.bind(SDRAGSTART,options.start):null;
208 				(options.drag)?$this.bind(SDRAGDRAG,options.drag):null;
209 				(options.end)?$this.bind(SDRAGEND,options.end):null;
210 							
211                 $this.bind(dragEvents.start, function(e){
212                    	handleDragEvent.call(this,e,options);
213                 });
214             }else{
215 
216 				(options.start)?$this.delegate(delegate,SDRAGSTART,options.start):null;
217 				(options.drag)?$this.delegate(delegate,SDRAGDRAG,options.drag):null;
218 				(options.end)?$this.delegate(delegate,SDRAGEND,options.end):null;	
219 				
220 				$this.delegate(delegate,dragEvents.start,function(e){
221 					handleDragEvent.call(this,e,options);
222 				})
223 			}
224         });
225         
226 		// Handler the event
227 		// "this" of this function will be the element
228         function handleDragEvent(e, options){
229 			//var $this = $(this);
230             
231             var $elem = $(this);
232             
233             var $document = $(document);
234             var id = "_" + snow.util.uuid(7);
235             
236             var dragStarted = false;
237             var startEvent = e;
238             var startPagePos = snow.event.eventPagePosition(startEvent);
239             
240             
241             
242             
243             // create the $helper if it is a draggable event.
244             var $helper; 
245 			
246 			
247 			// drag
248             $document.bind(dragEvents.drag + "." + id, function(e){
249             	
250             	// if the drag has not started, check if we need to start it
251             	if (!dragStarted){
252             		var currentPagePos = snow.event.eventPagePosition(e);
253             		
254             		// if the diff > threshold, then, we start the drag
255             		if (Math.abs(startPagePos.pageX - currentPagePos.pageY) > options.threshold){
256             			dragStarted = true;
257             			//create the sDragCtx
258             			$elem.data("sDragCtx",{});
259             			
260             			
261 						if (options.draggable === true){
262 							if ($.isFunction(options.helper)){
263 								$helper = $(options.helper.call($elem.get(0)));
264 							}else if (options.helper === "original"){
265 								
266 								$helper = $elem;
267 							}else if (options.helper === "clone"){
268 								$helper = $elem.clone();
269 								// make sure to remove the DOMElement ID
270 								$helper.attr("id",null);
271 								$helper.css("position","absolute");
272 								var elemPos = $elem.offset();
273 								$helper.css({
274 									top: elemPos.top,
275 									left: elemPos.left
276 								})
277 								//todo need to allow configurable helper parent (right now, it is the body)
278 								$("body").append($helper);					
279 							}
280 						}   
281 						var dragStartExtra = buildDragExtra(startEvent,$elem,$helper,SDRAGSTART);
282             			$elem.trigger(SDRAGSTART,[dragStartExtra]);
283             		}
284             	}
285             	
286             	if (dragStarted){
287 					var dragExtra = buildDragExtra(e,$elem,$helper,SDRAGDRAG);
288 					
289 	                var overElem;
290 	                if (options.draggable === true){
291 	                  overElem = findOverElement($helper,dragExtra);
292 	                  dragExtra.overElement = overElem;
293 	                }				
294 					
295 	            	$elem.trigger(SDRAGDRAG,[dragExtra]);
296 	            	
297 	            	if (options.draggable === true){
298 	            		moveElement($helper,dragExtra);
299 	            		var dropExtra = buildDropExtra($elem,$helper);
300 						triggerDropEventOnOverElement(SDRAGOVER,e,$elem,overElem,dropExtra);
301 					}
302 					
303 					// since we create "meta events" we consume this one	
304 	                e.preventDefault();
305 					e.stopPropagation();
306 				}
307             });
308             
309             // drag end
310             $document.bind(dragEvents.end + "." + id, function(e){
311             	if (dragStarted){
312 	                var extra = buildDragExtra(e,$elem,$helper,SDRAGEND);
313 	                var dropExtra; 
314 	                
315 	                var overElem;
316 	                if (options.draggable === true){
317 	                  overElem = findOverElement($helper,extra);
318 	                  extra.overElement = overElem;
319 	                }
320 	                 
321 	            	$elem.trigger(SDRAGEND,[extra]);
322 					
323 					// get the $overElem
324 					if (options.draggable === true){
325 						moveElement($helper,extra);
326 						dropExtra = buildDropExtra($elem,$helper);
327 						triggerDropEventOnOverElement(SDROP,e,$elem,overElem,dropExtra);
328 						
329 						if (!$helper.is($elem)){
330 							$helper.remove();
331 						}
332 					}				
333 					
334 					// delete the dragContext
335 					$elem.data("sDragCtx",null);
336 					
337 					// since we create "meta events" we consume this one
338 	                e.preventDefault();
339 					e.stopPropagation();
340 				}
341 				
342 				// unbind the document event
343 	            $(document).unbind(dragEvents.drag + "." + id);
344 	            $(document).unbind(dragEvents.end + "." + id);
345 					
346             });
347         }
348         
349     }

353     	var boxPos = $elem.position();
354 		$elem.css({
355 			left:boxPos.left + extra.deltaPageX,
356 			top:boxPos.top + extra.deltaPageY
357 		});

361     /**
362      * Trigger the drop event on the overElement
363      */
364     function triggerDropEventOnOverElement(eventType,event,$elem,overElem,dropExtra){
365     	var $overElem = $(overElem);
366     	
367     	//get the prevOverElem and do the enter/leave
368     	var sDragCtx = $elem.data("sDragCtx");
369     	var prevOverElem = sDragCtx.overElem;
370     	var sdragenterEvent, sdragleaveEvent;
371     	// if there are no prevOverElem then, we enter the new one
372     	if (typeof prevOverElem === "undefined" ){
373     		sdragenterEvent = $.Event(event);
374     		sdragenterEvent.target = overElem;
375     		sdragenterEvent.type = SDRAGENTER;
376     		$overElem.trigger(sdragenterEvent,dropExtra);
377     	}
378     	// if the new one and old one does not match, then, we need to leave the old elem and enter the new one
379     	else if (prevOverElem != overElem){
380     		//leave the old one
381     		sdragleaveEvent = $.Event(event);
382     		sdragleaveEvent.target = prevOverElem;
383     		sdragleaveEvent.type = SDRAGLEAVE;
384     		$(prevOverElem).trigger(sdragleaveEvent,dropExtra);
385     		
386     		//enter the new enter event
387     		sdragenterEvent = $.Event(event);
388     		sdragenterEvent.target = overElem;
389     		sdragenterEvent.type = SDRAGENTER;
390     		$overElem.trigger(sdragenterEvent,dropExtra);
391     		
392     	}
393     	
394     	sDragCtx.overElem = overElem;
395 		//create the event requested
396 		var sdragEvent = $.Event(event);
397 		sdragEvent.target = overElem;
398 		sdragEvent.type = eventType;
399 		$overElem.trigger(sdragEvent,dropExtra);    	
400     }

402     function findOverElement($elem,extra){
403     	$elem.hide();
404 		var overElem = document.elementFromPoint(extra.pageX,extra.pageY);
405 		$elem.show();
406 		return overElem;
407     }

409 	/**
410 	 * Build the extra event info for the drag event. 
411 	 */
412 	function buildDragExtra(event,$elem,$helper,dragType){
413 		snow.event.fixTouchEvent(event);
414 		var hasTouch = snow.ua.hasTouch();
415 		var extra = {
416 			eventSource: event,
417 			pageX: event.pageX,
418 			pageY: event.pageY			
419 		};
420 		
421 		if ($helper){
422 			extra.helperElement = $helper.get(0);
423 		}
424 		
425 		var oe = event.originalEvent;
426 		if (hasTouch){
427 			extra.touches = oe.touches;
428 		}
429 		
430 		var sDragCtx = $elem.data("sDragCtx");
431 		
432 		if (dragType === SDRAGSTART){
433 			sDragCtx.startPageX = extra.startPageX = extra.pageX;
434 			sDragCtx.startPageY = extra.startPageY = extra.pageY;
435 			
436 			sDragCtx.lastPageX = sDragCtx.startPageX = extra.startPageX;
437 			sDragCtx.lastPageY = sDragCtx.startPageY = extra.startPageY;
438 		}else if (dragType === SDRAGEND){
439 			// because, on iOs, the touchEnd event does not have the .touches[0].pageX
440 			extra.pageX = sDragCtx.lastPageX;
441 			extra.pageY = sDragCtx.lastPageY;
442 		}
443 		
444 		extra.startPageX = sDragCtx.startPageX;
445 		extra.startPageY = sDragCtx.startPageY;
446 		extra.deltaPageX = extra.pageX - sDragCtx.lastPageX;
447 		extra.deltaPageY = extra.pageY - sDragCtx.lastPageY;
448 		
449 		sDragCtx.lastPageX = extra.pageX;
450 		sDragCtx.lastPageY = extra.pageY;
451 		return extra;
452 	}

455 	 * Build the extra event info for the drop event
456 	 */
457 	function buildDropExtra($elem,$helper){
458 		var extra = {};
459 		extra.draggableElement = $elem.get(0);
460 		extra.helperElement = $helper.get(0);
461 		return extra;
462 	}

  3 /**
  4  * @namespace snow.dm data manager layers to register, access DAOs.
  5  * DAOs are javascript objects that must implement the following CRUD methods get, list, create, update, remove, and the getIdName method.<br /> 
  6  * Signatures of these methods should match the corresponding snow.dm.** methods.<br />
  7  * <br />
  8  * Note that DAO CRUD methods can return directly the result or a deferred object. Also, it is important to note that snow.dm.*** CRUD access methods
  9  * will always return deferred object (either the DAO return deferred, or a wrapped deferred if the DAO method did not return a deferred)<br />
 10  * <br />
 11  * The deferred pattern for daos allows the application to be agnostic about the call mode, synchronous or asynchronous (e.g. Ajax, Workers, and other callback based called), 
 12  * and consequently offer the maximum flexibility during development and production. It also enforce a good practice on how to build the UI components.<br />
 13  * <br />
 14  * If there is a need to access the daos result directly, the snow.sdm ("straight dm") can be used.  
 15  */

147 	/**
148 	 * DAO Interface: Create a new instance of the object for a give objectType and data. <br />
149 	 *
150 	 * The DAO should return or resolve with the newly created data.
151 	 *
152 	 * @param {Object} objectType
153 	 * @param {Object} data
154 	 */
155 	snow.sdm.create = function(objectType, data) {
156 		var result = getDao(objectType).create(objectType, data);
157 
158 		// if the result is a deferred object, then, wait until done to callChangeListeners
159 		if (result && $.isFunction(result.promise)) {
160 			result.done( function(newData) {
161 				callChangeListeners(objectType, "create", null, newData, data);
162 			});
163 
164 		} else {
165 			callChangeListeners(objectType, "create", null, result, data);
166 
167 		}
168 		return result;
169 
170 	};

215 	/**
216 	 * Private method to trigger the change(daoEvent) on all listeners.
217 	 * NOTE: this param names maps to what is daoChangeEvent
218 	 *
219 	 * @param {Object} objectType
220 	 * @param {Object} action ("remove" "create" "update")
221 	 * @param {Object} newData The data after the change (null if remove)
222 	 * @param {Object} saveData The data object that was used to do the create/update. Could be partial object.
223 	 */
224 	function callChangeListeners(objectType, action, newData, saveData) {
225 		var listeners = daoChangeEventListeners[objectType], listener;
226 
227 		var daoChangeEvent = {
228 			objectType: objectType,
229 			action: action,
230 			newData: newData,
231 			saveData: saveData
232 		};
233 
234 		if (listeners) {
235 			for (var i = 0; i < listeners.length; i++) {
236 				listeners[i](daoChangeEvent);
237 			}
238 		}
239 	};

241 	/**
242 	 * Wrap with a deferred object if the obj is not a deferred itself.
243 	 */
244 	function wrapWithDeferred(obj) {
245 		//if it is a deferred, then, trust it, return it.
246 		if (obj && $.isFunction(obj.promise)) {
247 			return obj;
248 		} else {
249 			var dfd = $.Deferred();
250 			dfd.resolve(obj);
251 			return dfd;
252 		}
253 	}

279 	SimpleDao.prototype.get = function(objectType, id) {
280 		var idx = snow.util.array.getIndex(this._store, "id", id);
281 		return this._store[idx];
282 	}

284 	//for now, just support opts.orderBy
285 	SimpleDao.prototype.list = function(objectType, opts) {
286 		//TODO: probably need to copy the array to avoid giving the original array
287 		var resultSet = this._store;
288 
289 		if (opts) {
290 			if (opts.orderBy) {
291 				resultSet = snow.util.array.sortBy(resultSet, opts.orderBy)
292 			}
293 			if (opts.match) {
294 				resultSet = $.map(resultSet, function(val, idx) {
295 					var pass = true;
296 
297 					$.each(opts.match, function(k, v) {
298 						if (val[k] === v) {
299 							pass = pass && true;
300 						} else {
301 							pass = false;
302 						}
303 					});
304 
305 					return (pass) ? val : null;
306 				});
307 
308 			}
309 		}
310 		return resultSet;
311 	}
312 
313 	SimpleDao.prototype.create = function(objectType, data) {
314 		var idName = snow.dm.getIdName(objectType);
315 
316 		// if the id has already been created, no biggies, otherwise, create it.
317 		if (typeof data[idName] !== "undefined") {
318 			var er = "SimpleDao.create error: Id present in data object. Cannot create. You might want to call update instead.";
319 			snow.log.debug(er);
320 			throw er;
321 
322 		}
323 
324 		data[idName] = snow.util.uuid(12);
325 
326 		this._store.push(data);
327 
328 		return data;
329 	}
330 
331 	SimpleDao.prototype.update = function(objectType, id, data) {
332 		// if there is an id, make sure it matches
333 		var idName = snow.dm.getIdName(objectType);
334 		var dataId = data[idName];
335 		if (typeof dataId !== "undefined" && dataId != id) {
336 			var er = "SimpleDao.update error: Id in param and data does not match. Cannot update data.";
337 			snow.log.debug(er);
338 			throw er;
339 		}
340 
341 		// remove the id from the data to be saved, not needed.
342 		delete data[idName];
343 
344 		//get the data, and populate the new value
345 		var storeData = this.get(objectType, id);
346 		if (storeData) {
347 			$.extend(storeData, data);
348 			return storeData;
349 		}
350 	}
351 
352 	SimpleDao.prototype.remove = function(objectType, id) {
353 		var oldData = this.get(objectType, id);
354 		var idx = snow.util.array.getIndex(this._store, "id", id);
355 		if (idx > -1) {
356 			snow.util.array.remove(this._store, idx);
357 		} else {
358 			var er = "SimpleDao.remove error: Ojbect " + objectType + "[" + id + "] not found. Cannot delete.";
359 			snow.log.debug(er);
360 			throw er;
361 		}
362 
363 		return oldData;
364 
365 	}

385 	function SimpleRelDao(store, rels) {
386 		this._super.init.call(this, store);
387 		this._rels = rels;
388 		this._relDic = {};
389 		for (var i = 0, l = rels.length; i < l; i++) {
390 			this._relDic[rels[i]] = rels[i] + "_id";
391 		}

396 	SimpleRelDao.prototype.get = function(objectType, id) {
397 		var result = this._super.get.call(this, objectType, id);
398 		return completeData.call(this, result);
399 	}
400 
401 	SimpleRelDao.prototype.list = function(objectType, opts) {
402 		var resultSet = this._super.list.call(this, objectType, opts);
403 
404 		// Now, go through the list, adn load the other object type.
405 		if (this._rels) {
406 			var dao = this;
407 			$.each(resultSet, function(idx, val) {
408 				completeData.call(dao, val);
409 			});
410 
411 		}
412 
413 		return resultSet;
414 
415 	}

417 	SimpleRelDao.prototype.save = function(objectType, data) {
418 		// make sure to remove the direct object reference (we expect the rel_id at this point)
419 		// TODO: probably need to extra the id from the object reference in case there is now rel_id
420 		if (this._rels) {
421 			var tmpPropName;
422 			for (var i = 0, l = this._rels.length; i < l; i++) {
423 				tmpPropName = this._rels[i];
424 				if (typeof data[tmpPropName] !== "undefined") {
425 					delete data[tmpPropName];
426 				}
427 			}
428 		}
429 
430 		var result = this._super.save.call(this, objectType, data);
431 		return completeData.call(this, result);
432 	}

434 	// ------ Private Helpers ------ //
435 	// complete the data with the related objects
436 	function completeData(val) {
437 		var rels = this._rels;
438 		var rel, propIdName, obj, objId;
439 		for (var i = 0; i < rels.length; i++) {
440 			rel = rels[i];
441 			propIdName = this._relDic[rel];
442 			objId = val[propIdName];
443 			if (typeof objId !== "undefined") {
444 				obj = snow.dm.get(rel, objId);
445 				if (typeof obj !== "undefined" && obj != null) {
446 					val[rel] = obj;
447 				}
448 			}
449 		}
450 		return val;
451 	}

468 	/**
469 	 * Return the objRef {id,type,$element} (or a list of such) of the closest html element matching the objType match the data-obj_type.<br />
470 	 * If no objType, then, return the first objRef of the closest html element having a data-obj_type. <br />
471 	 *
472 	 * @param {String} objType (optional) the object table
473 	 * @return null if not found, single object with {id,type,$element} if only one jQuery object, a list of such if this jQuery contain multiple elements.
474 	 */
475 	$.fn.sObjRef = function(objType) {
476 		var resultList = [];
477 
478 		var obj = null;
479 		// iterate and process each matched element
480 		this.each( function() {
481 			var $this = $(this);
482 			var $sObj;
483 			if (objType) {
484 				$sObj = $this.closest("[data-obj_type='" + objType + "']");
485 			} else {
486 				$sObj = $this.closest("[data-obj_type]");
487 			}
488 			if ($sObj.length > 0) {
489 				var objRef = {
490 					type: $sObj.attr("data-obj_type"),
491 					id: $sObj.attr("data-obj_id"),
492 					$element: $sObj
493 				}
494 				resultList.push(objRef);
495 			}
496 		});
497 
498 		if (resultList.length === 0) {
499 			return null;
500 		} else if (resultList.length === 1) {
501 			return resultList[0];
502 		} else {
503 			return resultList;
504 		}
505 
506 	};
